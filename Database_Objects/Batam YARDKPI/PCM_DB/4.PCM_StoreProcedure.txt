DROP PROC  [dbo].[sp_pcm_del]
CREATE PROC [dbo].[sp_pcm_del]
    (
        @Parameter1 nvarchar(30)
    )
AS
BEGIN
    IF @Parameter1 = 'welder_details'
        BEGIN
		
       update dbo.dim_entp_pcm_welder_activity
       set active = 0
       from dbo.dim_entp_pcm_welder_activity a,
            yardkpi.stage_entp_pcm_Welder_activity b
	   where
	   a.fab_code = b.fab_code and 
       a.wctrl=b.wctrl and
       a.weld_suffix=b.weld_suffix and
       a.stress_relief=b.stress_relief and
       a.weld_repair_ref=b.weld_repair_ref and
       a.welder=b.welder and
       a.weld_procedure=b.weld_procedure and
       a.welding_process=b.welding_process
	   and a.active = 1;
            
        END
		
		
    IF @Parameter1 = 'ndt_cert_details'
        BEGIN

	   update dbo.dim_entp_pcm_certification_package
       set active = 0
       from dbo.dim_entp_pcm_certification_package a,
            yardkpi.stage_entp_pcm_certification_package b
       where    	
		a.fab_code = b.fab_code and 
		a.ndt_drawing=b.ndt_drawing and
		a.job=b.job and
		a.item=b.item  
		and a.active = 1;
            
        END
		
	IF @Parameter1 = 'ndt_details'
        BEGIN
		
       update dbo.dim_entp_pcm_ndt_details
       set active = 0
       from dbo.dim_entp_pcm_ndt_details a,
            yardkpi.stage_entp_pcm_ndt_details b
	   where    	
		a.fab_code = b.fab_code and 
		a.wctrl=b.wctrl and
		a.weld_suffix=b.weld_suffix and
		a.stress_relief=b.stress_relief and
		a.weld_repair_ref=b.weld_repair_ref and
		a.ndt_type=b.ndt_type 
		and a.active = 1;
	  
            
        END
		
		
    IF @Parameter1 = 'Paint_details'
        BEGIN
              	       
					   
	   update dbo.dim_entp_pcm_paint_detail
       set active = 0
       from dbo.dim_entp_pcm_paint_detail a,
            yardkpi.stage_entp_pcm_paint_detail b
	   where    	
		a.fab_code = b.fab_code and 
		a.pctrl=b.pctrl
		and a.active = 1		;
            
        END
		
		
	IF @Parameter1 = 'weld_details'
        BEGIN
             	       
	   update dbo.dim_entp_pcm_Weld_details
       set active = 0
       from dbo.dim_entp_pcm_Weld_details a,
            yardkpi.stage_entp_pcm_Weld_details b
       where    	
		a.fab_code = b.fab_code and 
		a.wctrl=b.wctrl and
		a.weld_suffix=b.weld_suffix and
		a.stress_relief=b.stress_relief and
		a.weld_repair_ref=b.weld_repair_ref
		and a.active = 1;
            
        END


END
