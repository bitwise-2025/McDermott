CREATE VIEW [yardkpi].[stage_entp_gmps_contract_vw]
AS select ROW_NUMBER() over(ORDER BY contract ASC) as rowID,
source_code,
sub_source_code,
contract,
contract_title,
contract as project_number,
LastUpdate
from (
select distinct
    'SYSTEM' as source_code,
    'GMPS' as sub_source_code,
    contract_no as contract ,
	contract_no as contract_title,
	contract_no as project_number,
    getdate() LastUpdate
from [yardkpi].[stage_gmps_sp_contract] dcsr
where not exists (select 1 from dbo.dim_ent_contract
        where contract_number = dcsr.contract_no
        and active = 1)		
        ) disa;
