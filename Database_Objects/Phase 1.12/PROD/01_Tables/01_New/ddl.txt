create table generic_json_header (
load_name			varchar(240) not null,
load_description		varchar(500) not null,
source_table_name		varchar(500) not null,
target_view_name		varchar(500) not null,
error_target_view_name		varchar(500) not null,
error_column_not_configured	varchar(1)   not null,
active_flag			varchar(1)   not null);

create index ind1 on generic_json_header(load_name);

create table generic_json_detail (
gjd_id				int identity(1,1) not null,
load_name			varchar(240) not null,
json_column_name		varchar(500) not null,
json_column_data_type		varchar(120) not null,
json_column_data_group		varchar(2) not null, -- V, N, D, DT
json_column_data_group_format	varchar(120),
json_column_alias_name		varchar(120),
active_flag			varchar(1));

create index ind1 on generic_json_detail (load_name);

create table generic_json_errors_data (
gjed_id				int identity(1,1) not null,
load_name			varchar(240) not null,
json_attribute 			varchar(8000),
process_datetime		datetime,
history_flag			varchar(1));
