
Create PROC [EUD].[sp_EUT_Aspentech_exl]
AS
BEGIN
DECLARE @STMT AS NVARCHAR(MAX) ='
INSERT INTO   EUD.EUT_exl
SELECT 
Vendor,[Date],Product,[User],Gross,[Tokens Consumed by License],Duration,usageType,zSourceFileName,zCreatedDateTime   
FROM [EUD].[EUT_v] where zSourceFileName like ''%Aspentech%'''
 EXEC sp_executesql @STMT;
 END

Create PROC [eud].[sp_EUT_AVEVA_exl]
AS
BEGIN
DECLARE @STMT AS NVARCHAR(MAX) ='
DELETE FROM eud.EUT_exl WHERE SourceFileName =(SELECT DISTINCT zSourceFileName FROM [eud].[EUT_v]);
INSERT INTO   eud.EUT_exl
SELECT 
Vendor,[Usage_Date],Usage_Log_App,[User],Gross,[Tokens Consumed by License],Duration,usageType,zSourceFileName,zCreatedDateTime   
FROM [eud].[EUT_v] where zSourceFileName like ''%AVEVA%'''
 EXEC sp_executesql @STMT;

Create PROC [dbo].[sp_EUT_Aveva_exl]
AS
BEGIN
DECLARE @STMT AS NVARCHAR(MAX) ='
DELETE FROM dbo.EUT_exl WHERE SourceFileName =(SELECT DISTINCT File_Name FROM [dbo].[EUT_txt_Aveva]);
INSERT INTO   dbo.EUT_exl
SELECT 
Vendor,Usage_DT,Application,User_Name,''0'' As Gross, ''0'' As token ,''0'' As Duration,NULL as usagetype,File_Name,CreatedDateTime 
FROM [dbo].[EUT_txt_Aveva] where File_Name like ''%Aveva%'''
 EXEC sp_executesql @STMT;
 END
