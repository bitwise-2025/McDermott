
Create VIEW [eud].[EUT_v]
AS
SELECT
rssk,Vendor,usageDate,productName,userName,usageType,tokensConsumed,[Usage Date],Application,Persona,Gross,Start,App,[User],[Date],Product,[Tokens Consumed by License],Duration,Usage_Date,Usage_Log_App,zSourceFileName,
zCreatedDateTime,zVersionNo
FROM (
		SELECT rssk,  [KEY] as [colName], value AS RowValue  
		FROM [eud].[EUT_JSON] 
			CROSS APPLY OPENJSON(JSON_Attribute)
		) as j PIVOT (MAX(RowValue) for  [colName] in ( 
Vendor,usageDate,productName,userName,usageType,tokensConsumed,[Usage Date],Application,Persona,Gross,Start,App,[User],[Date],Product,[Tokens Consumed by License],Duration,Usage_Date,Usage_Log_App,zSourceFileName,
zCreatedDateTime,zVersionNo		
		)) as p;

CREATE VIEW [EUD].[dim_eut_vendor_source_vw] as

SELECT

b.dim_eut_vendor_nk As dim_eut_vendor_nk ,

c.dim_eut_vendor_app_nk As dim_eut_vendor_app_nk ,

a.Vendor As Vendor ,

CONVERT(DATETIME,a.usage_Date,101) As usage_Date,

a.Application As Application ,

CONCAT(Replace(a.Network_ID,'@mcdermott.com',''),'@mcdermott.com') as Network_ID,

a.Gross As Gross,

a.token As token ,

a.Duration As Duration,

a.SourceFileName As SourceFileName,

a.CreatedDateTime As CreatedDateTime

 

FROM EUD.EUT_exl a

 LEFT JOIN EUD.dim_eut_vendor b on a.vendor=b.vendor

 LEFT JOIN EUD.dim_eut_vendor_app c on a.vendor=c.vendor AND a.Application=c.Application
